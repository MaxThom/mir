MCP Library Code Comparison

Same tool — list_devices(namespace string) — implemented in both libraries

A — mark3labs/mcp-go
// Server setup
s := server.NewMCPServer("Echo", "1.0.0",
    server.WithToolCapabilities(false),
)

// Register tool — manual schema
tool := mcp.NewTool("list_devices",
    mcp.WithDescription("List Mir devices"),
    mcp.WithString("namespace",
        mcp.Description("Filter by namespace"),
    ),
)
s.AddTool(tool, e.handleListDevices)

// Mount on existing mux
sseServer := server.NewSSEServer(s,
    server.WithStaticBasePath("/mcp"),
    server.WithBaseURL("http://localhost:8080"),
)
mux.Handle("/mcp/", sseServer)

// Handler — manual param extraction
func (e *EchoServer) handleListDevices(
    ctx context.Context,
    req mcp.CallToolRequest,
) (*mcp.CallToolResult, error) {
    ns := req.GetString("namespace", "")

    devices, err := e.listDevices(ctx, ns)
    if err != nil {
        return mcp.NewToolResultError(err.Error()), nil
    }
    out, _ := json.Marshal(devices)
    return mcp.NewToolResultText(string(out)), nil
}
B — modelcontextprotocol/go-sdk  ★ Official
// Server setup
s := mcp.NewServer(&mcp.Implementation{
    Name: "Echo", Version: "1.0.0",
}, nil)

// Register tool — schema from struct
type ListDevicesInput struct {
    Namespace string `json:"namespace" jsonschema:"Filter by namespace"`
}
mcp.AddTool(s,
    &mcp.Tool{
        Name:        "list_devices",
        Description: "List Mir devices",
    },
    e.handleListDevices,
)

// Mount on existing mux
handler := mcp.NewSSEHandler(
    func(r *http.Request) *mcp.Server { return s },
    nil,
)
mux.Handle("/mcp/", handler)

// Handler — typed input, no manual parsing
func (e *EchoServer) handleListDevices(
    ctx context.Context,
    req *mcp.CallToolRequest,
    input ListDevicesInput,
) (*mcp.CallToolResult, any, error) {
    devices, err := e.listDevices(ctx, input.Namespace)
    if err != nil {
        return nil, nil, err
    }
    out, _ := json.Marshal(devices)
    return &mcp.CallToolResult{
        Content: []mcp.Content{
            &mcp.TextContent{Text: string(out)},
        },
    }, nil, nil
}
Key difference
mark3labs: you declare the schema manually with mcp.WithString(...), then extract values manually with req.GetString("key", ""). More explicit, slightly more boilerplate per tool.
official SDK: you define a typed Go struct — the library infers the JSON schema automatically. The handler receives a fully-typed, already-parsed input. Cleaner, less error-prone at scale.
